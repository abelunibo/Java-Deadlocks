166 class RayTracerRunner extends RayTracer implements Runnable {
167 
168     int id, height, width;
169     TournamentBarrier br;
170 
171     public RayTracerRunner(int id, int width, int height, TournamentBarrier br) {
172         this.id = id;
173         this.width = width;
174         this.height = height;
175         this.br = br;
176 
177         JGFInstrumentor.startTimer("Section3:RayTracer:Init");
178 
179         // create the objects to be rendered
180         scene = createScene();
181 
182         // get lights, objects etc. from scene.
183         setScene(scene);
184 
185         numobjects = scene.getObjects();
186         JGFRayTracerBench.staticnumobjects = numobjects;
187 
188         JGFInstrumentor.stopTimer("Section3:RayTracer:Init");
189 
190     }
191 
192     public void run() {
193 
194         // Set interval to be rendered to the whole picture
195         // (overkill, but will be useful to retain this for parallel versions)
196 
197         Interval interval = new Interval(0, width, height, 0, height, 1, id);
198 
199         // synchronise threads and start timer
200 
201         br.DoBarrier(id);
202         if (id == 0)
203             JGFInstrumentor.startTimer("Section3:RayTracer:Run");
204 
205         render(interval);
206 
207         // Signal this thread has done iteration
208 
209         synchronized (scene) {
210             for (int i = 0; i < JGFRayTracerBench.nthreads; i++)
211                 if (id == i)
212                     JGFRayTracerBench.checksum1 = JGFRayTracerBench.checksum1
213                             + checksum;
214         }
215 
216         // synchronise threads and stop timer
217 
218         br.DoBarrier(id);
219         if (id == 0)
220             JGFInstrumentor.stopTimer("Section3:RayTracer:Run");
221 
222     }
223 }
