25  package raytracer;
26  
27  
28  public class Sphere extends Primitive implements java.io.Serializable {
29    Vec      c;
30    double   r, r2;
31    Vec      v,b; // temporary vecs used to minimize the memory load
32  
33    public Sphere(Vec center, double radius) {
34      c = center;
35      r = radius;
36      r2 = r*r;
37      v=new Vec();
38      b=new Vec();
39    }
40    
41    public Isect intersect(Ray ry) {
42      double b, disc, t;
43      Isect ip;
44      v.sub2(c, ry.P);
45      b = Vec.dot(v, ry.D);
46      disc = b*b - Vec.dot(v, v) + r2;
47      if (disc < 0.0) {
48        return null;
49      }
50      disc = Math.sqrt(disc);
51      t = (b - disc < 1e-6) ? b + disc : b - disc;
52      if (t < 1e-6) {
53        return null;
54      }
55      ip = new Isect();
56      ip.t = t;
57      ip.enter = Vec.dot(v, v) > r2 + 1e-6 ? 1 : 0;
58      ip.prim = this;
59      ip.surf = surf;
60      return ip;
61    }
62  
63    public Vec normal(Vec p) {
64      Vec r;
65      r = Vec.sub(p, c);
66      r.normalize();
67      return r;
68    }
69  
70    public String toString() {
71      return "Sphere {" + c.toString() + "," + r + "}";
72    }
73      
74    public Vec getCenter() {
75      return c;
76    }
77    public void setCenter(Vec c) {
78      this.c = c;
79    }
80  }
