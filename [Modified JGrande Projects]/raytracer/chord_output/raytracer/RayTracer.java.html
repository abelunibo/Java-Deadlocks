95    Scene createScene() {
96      int x = 0;
97      int y = 0;
98      
99      Scene scene = new Scene();
100    
101 
102     /* create spheres */ 
103 
104     Primitive p; 
105     int nx = 4;
106     int ny = 4;
107     int nz = 4;
108       for (int i = 0; i<nx; i++) {
109     for (int j = 0; j<ny; j++) {
110       for (int k = 0; k<nz; k++) {
111               double xx = 20.0 / (nx - 1) * i - 10.0; 
112               double yy = 20.0 / (ny - 1) * j - 10.0; 
113               double zz = 20.0 / (nz - 1) * k - 10.0; 
114           
115           p = new Sphere(new Vec(xx,yy,zz), 3);
116           //p.setColor(i/(double) (nx-1), j/(double)(ny-1), k/(double) (nz-1));
117           p.setColor(0, 0, (i+j)/(double) (nx+ny-2));
118           p.surf.shine = 15.0;
119           p.surf.ks = 1.5 - 1.0;
120           p.surf.kt= 1.5 - 1.0;
121           scene.addObject(p);
122       }
123     }
124       }
125   
126 
127       
128     /* Creates five lights for the scene */
129     scene.addLight(new Light(100, 100, -50, 1.0));
130     scene.addLight(new Light(-100, 100, -50, 1.0));
131     scene.addLight(new Light(100,-100,-50, 1.0));
132     scene.addLight(new Light(-100,-100,-50, 1.0));
133     scene.addLight(new Light(200, 200, 0, 1.0));
134 
135     /* Creates a View (viewing point) for the rendering scene */
136     View v = new View(  new Vec(x, 20, -30),
137                         new Vec(x, y, 0),
138                         new Vec(0, 1, 0),
139                         1.0,
140             35.0 * 3.14159265 / 180.0,
141                         1.0);
142 /*
143     v.from = new Vec(x, y, -30);
144     v.at = new Vec(x, y, -15);
145     v.up = new Vec(0, 1, 0);
146     v.angle = 35.0 * 3.14159265 / 180.0;
147     v.aspect = 1.0; 
148     v.dist = 1.0;
149     
150     */
151     scene.setView(v);
152     
153     return scene;
154   }

157 public void setScene(Scene scene)
158     {
159     // Get the objects count
160     int nLights = scene.getLights();
161     int nObjects = scene.getObjects();
162     
163     lights = new Light[nLights];
164     prim = new Primitive[nObjects];
165     
166     // Get the lights
167     for (int l = 0; l < nLights; l++) {
168       lights[l]=scene.getLight(l);
169     }
170 
171     // Get the primitives
172     for (int o = 0; o < nObjects; o++) {
173       prim[o]=scene.getObject(o);
174     }
175     
176     // Set the view
177     view = scene.getView();
178   }

180 public void render(Interval interval)
181     {  
182   
183   // Screen variables
184     int row[] = new int[interval.width * (interval.yto-interval.yfrom)];
185     int pixCounter=0; //iterator
186    
187  
188   // Rendering variables
189     int x, y, red, green, blue;
190     double xlen, ylen;    
191     Vec viewVec;
192     
193 
194     viewVec = Vec.sub(view.at, view.from);
195     
196     viewVec.normalize();
197     
198     Vec tmpVec = new Vec(viewVec);
199     tmpVec.scale(Vec.dot(view.up, viewVec));
200     
201     Vec upVec = Vec.sub(view.up, tmpVec);
202     upVec.normalize();
203       
204     Vec leftVec = Vec.cross(view.up, viewVec);
205     leftVec.normalize();
206 
207     double frustrumwidth = view.dist * Math.tan(view.angle);
208       
209     upVec.scale(-frustrumwidth);
210     leftVec.scale(view.aspect * frustrumwidth);
211       
212     Ray r = new Ray(view.from, voidVec);    
213     Vec col = new Vec();
214       
215     // Header for .ppm file 
216     // System.out.println("P3"); 
217     // System.out.println(width + " " + height);
218     // System.out.println("255"); 
219  
220 
221     // All loops are reversed for 'speedup' (cf. thinking in java p331)
222     
223     // For each line
224 
225     for(y = interval.yfrom+interval.threadid; y < interval.yto; y+=JGFRayTracerBench.nthreads) {
226 
227       ylen = (double)(2.0 * y) / (double)interval.width - 1.0;
228       // System.out.println("Doing line " + y);
229       // For each pixel of the line
230       for(x = 0; x < interval.width; x++) {
231     xlen = (double)(2.0 * x) / (double)interval.width - 1.0;
232     r.D = Vec.comb(xlen, leftVec, ylen, upVec);
233     r.D.add(viewVec);
234     r.D.normalize();
235     col = trace(0, 1.0, r);
236     
237     // computes the color of the ray
238     red = (int)(col.x * 255.0);
239     if (red > 255)
240       red = 255;
241     green = (int)(col.y * 255.0);
242     if (green > 255)
243       green = 255;
244     blue = (int)(col.z * 255.0);
245     if (blue > 255)
246       blue = 255;
247 
248         checksum += red;
249         checksum += green;
250         checksum += blue;
251 
252     // RGB values for .ppm file 
253         // System.out.println(red + " " + green + " " + blue); 
254     // Sets the pixels
255     row[pixCounter++] =  alpha | (red << 16) | (green << 8) | (blue);
256       } // end for (x)
257     } // end for (y)
258 
259     }

261   boolean intersect(Ray r, double maxt) {
262     Isect tp;
263     int i, nhits;
264     
265     nhits = 0;
266     inter.t = 1e9;
267     for(i = 0; i < prim.length; i++) {
268       // uses global temporary Prim (tp) as temp.object for speedup
269       tp = prim[i].intersect(r);
270       if (tp != null && tp.t < inter.t) {
271     inter.t = tp.t;
272     inter.prim = tp.prim;
273     inter.surf = tp.surf;
274     inter.enter = tp.enter;
275     nhits++;
276       }
277     }
278     return nhits > 0 ? true : false;
279   }

281   /**
282    * Checks if there is a shadow
283    * @param r The ray
284    * @return Returns 1 if there is a shadow, 0 if there isn't
285    */
286   int Shadow(Ray r, double tmax) {
287     if (intersect(r, tmax))
288       return 0;
289     return 1;
290   }

293   /**
294    * Return the Vector's reflection direction
295    * @return The specular direction
296    */
297   Vec SpecularDirection(Vec I, Vec N) {
298     Vec r;
299     r = Vec.comb(1.0/Math.abs(Vec.dot(I, N)), I, 2.0, N);
300     r.normalize();
301     return r;
302   }

304   /**
305    * Return the Vector's transmission direction
306    */
307   Vec TransDir(Surface m1, Surface m2, Vec I, Vec N) {
308     double n1, n2, eta, c1, cs2;
309     Vec r;
310     n1 = m1 == null ? 1.0 : m1.ior;
311     n2 = m2 == null ? 1.0 : m2.ior;
312     eta = n1/n2;
313     c1 = -Vec.dot(I, N);
314     cs2 = 1.0 - eta * eta * (1.0 - c1 * c1);
315     if (cs2 < 0.0)
316       return null;
317     r = Vec.comb(eta, I, eta * c1 - Math.sqrt(cs2), N);
318     r.normalize();
319     return r;
320   }

323   /**
324    * Returns the shaded color
325    * @return The color in Vec form (rgb)
326    */
327   Vec shade(int level, double weight, Vec P, Vec N, Vec I, Isect hit) {
328     double n1, n2, eta, c1, cs2;
329     Vec r;
330     Vec tcol;
331     Vec R;
332     double t, diff, spec;
333     Surface surf;
334     Vec col;
335     int l;
336     
337     col = new Vec();
338     surf = hit.surf;
339     R = new Vec();
340     if (surf.shine > 1e-6) {
341       R = SpecularDirection(I, N);
342     }
343 
344     // Computes the effectof each light
345     for(l = 0; l < lights.length; l++) {
346       L.sub2(lights[l].pos, P);
347       if (Vec.dot(N, L) >= 0.0) {
348     t = L.normalize();
349 
350     tRay.P=P;
351     tRay.D=L;
352 
353     // Checks if there is a shadow
354     if (Shadow(tRay, t) > 0) {
355       diff = Vec.dot(N, L) * surf.kd *
356         lights[l].brightness;
357 
358       col.adds(diff,surf.color);
359       if (surf.shine > 1e-6) {
360         spec = Vec.dot(R, L);
361         if (spec > 1e-6) {
362           spec = Math.pow(spec, surf.shine);
363           col.x += spec;
364           col.y += spec;
365           col.z += spec;
366         }
367       }
368     }
369       } // if
370     } // for
371     
372     tRay.P=P;
373     if (surf.ks * weight > 1e-3) {
374       tRay.D = SpecularDirection(I, N);
375       tcol = trace(level + 1, surf.ks * weight, tRay);
376       col.adds(surf.ks, tcol);
377     }
378     if (surf.kt * weight > 1e-3) {
379       if (hit.enter > 0)
380     tRay.D = TransDir(null, surf, I, N);
381       else
382     tRay.D = TransDir(surf, null, I, N);
383       tcol = trace(level + 1, surf.kt * weight, tRay);
384       col.adds(surf.kt, tcol);
385     }
386 
387     // garbaging...
388     tcol=null;
389     surf=null;
390 
391     return col;
392   }

394   /**
395    * Launches a ray
396    */
397   Vec trace (int level, double weight, Ray r) {
398     Vec P, N;
399     boolean hit;
400     
401     // Checks the recursion level
402     if (level > 6) {
403       return new Vec();
404     }
405     
406     hit = intersect(r, 1e6);
407     if (hit) {
408       P = r.point(inter.t);
409       N = inter.prim.normal(P);
410       if (Vec.dot(r.D, N) >= 0.0) {
411     N.negate();
412       }
413       return shade(level, weight, P, N, r.D, inter);
414     }
415     // no intersection --> col = 0,0,0
416     return voidVec;
417   }
