25  package raytracer;
26  
27  
28  /**
29   * This class reflects the 3d vectors used in 3d computations
30   */
31  public class Vec implements java.io.Serializable {
32  
33    /**
34     * The x coordinate
35     */
36    public double x; 
37  
38    /**
39     * The y coordinate
40     */
41    public double y;
42  
43    /**
44     * The z coordinate
45     */
46    public double z;
47  
48    /**
49     * Constructor
50     * @param a the x coordinate
51     * @param b the y coordinate
52     * @param c the z coordinate
53     */
54    public Vec(double a, double b, double c) {
55      x = a;
56      y = b;
57      z = c;
58    }
59  
60    /**
61     * Copy constructor
62     */
63    public Vec(Vec a) {
64      x = a.x;
65      y = a.y;
66      z = a.z;
67    }
68    /**
69     * Default (0,0,0) constructor
70     */
71    public Vec() {
72      x = 0.0;
73      y = 0.0; 
74      z = 0.0;
75    }
76  
77    /**
78     * Add a vector to the current vector
79     * @param: a The vector to be added
80     */
81    public final void add(Vec a) {
82      x+=a.x;
83      y+=a.y;
84      z+=a.z;
85    }  
86  
87    /**
88     * adds: Returns a new vector such as
89     * new = sA + B
90     */
91    public static Vec adds(double s, Vec a, Vec b) {
92      return new Vec(s * a.x + b.x, s * a.y + b.y, s * a.z + b.z);
93    }
94      
95    /**
96     * Adds vector such as:
97     * this+=sB
98     * @param: s The multiplier
99     * @param: b The vector to be added
100    */
101   public final void adds(double s,Vec b){
102       x+=s*b.x;
103       y+=s*b.y;
104       z+=s*b.z;
105   }
106 
107   /**
108    * Substracs two vectors
109    */
110   public static Vec sub(Vec a, Vec b) {
111     return new Vec(a.x - b.x, a.y - b.y, a.z - b.z);
112   }
113 
114   /**
115    * Substracts two vects and places the results in the current vector
116    * Used for speedup with local variables -there were too much Vec to be gc'ed
117    * Consumes about 10 units, whether sub consumes nearly 999 units!! 
118    * cf thinking in java p. 831,832
119    */
120   public final void sub2(Vec a,Vec b) {
121     this.x=a.x-b.x;
122     this.y=a.y-b.y;
123     this.z=a.z-b.z;
124   }
125 
126   public static Vec mult(Vec a, Vec b) {
127     return new Vec(a.x * b.x, a.y * b.y, a.z * b.z);
128   }
129 
130   public static Vec cross(Vec a, Vec b) {
131     return
132       new Vec(a.y*b.z - a.z*b.y,
133           a.z*b.x - a.x*b.z,
134           a.x*b.y - a.y*b.x);
135   }
136 
137   public static double dot(Vec a, Vec b) {
138     return a.x*b.x + a.y*b.y + a.z*b.z;
139   }
140 
141   public static Vec comb(double a, Vec A, double b, Vec B) {
142     return
143       new Vec(a * A.x + b * B.x,
144           a * A.y + b * B.y,
145           a * A.z + b * B.z);
146   }
147 
148   public final void comb2(double a,Vec A,double b,Vec B) {
149     x=a * A.x + b * B.x;
150     y=a * A.y + b * B.y;
151     z=a * A.z + b * B.z;      
152   }
153 
154   public final void scale(double t) {
155     x *= t;
156     y *= t;
157     z *= t;
158   }
159 
160   public final void negate() {
161     x = -x;
162     y = -y;
163     z = -z;
164   }
165 
166   public final double normalize() {
167     double len;
168     len = Math.sqrt(x*x + y*y + z*z);
169     if (len > 0.0) {
170       x /= len;
171       y /= len;
172       z /= len;
173     }
174     return len;
175   }
176 
177   public final String toString() {
178     return "<" + x + "," + y + "," + z + ">";
179   }
180 }
