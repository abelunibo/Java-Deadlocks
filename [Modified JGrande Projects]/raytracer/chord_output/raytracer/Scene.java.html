25  package raytracer;
26  
27  
28  import java.util.Vector;
29  
30  public class Scene implements java.io.Serializable
31  {
32      public final Vector lights;
33      public final Vector objects;
34      private View view;  
35    
36      public Scene ()
37      {
38          this.lights = new Vector ();
39          this.objects = new Vector ();        
40      }
41    
42      public void addLight(Light l)
43      {
44          this.lights.addElement(l);
45      }
46    
47      public void addObject(Primitive object)
48      {
49          this.objects.addElement(object);
50      }
51    
52      public void setView(View view)
53      {
54          this.view = view;
55      }
56    
57      public View getView()
58      {
59          return this.view;
60      }
61    
62      public Light getLight(int number)
63      {
64          return (Light) this.lights.elementAt(number);
65      }
66    
67      public Primitive getObject(int number)
68      {
69          return (Primitive) objects.elementAt(number);
70      }
71    
72      public int getLights()
73      {
74          return this.lights.size();
75      }
76    
77      public int getObjects()
78      {
79          return this.objects.size();
80      }
81    
82      public void setObject(Primitive object, int pos)
83      {
84          this.objects.setElementAt(object, pos);
85      }
86  }
